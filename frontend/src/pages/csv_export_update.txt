const handleExportCSV = () => {
    const now = new Date();
    const dateStr = now.toISOString().split('T')[0];
    const timeStr = now.toLocaleTimeString();
    const userName = localStorage.getItem('user_name') || 'User';
    
    const csvLines = [
      'RESEARCH EXPENSE TRACKER - EXPENSE REPORT',
      `Generated: ${dateStr} at ${timeStr}`,
      `Generated by: ${userName}`,
      '',
      'EXPENSE SUMMARY',
      `Total Expenses,BDT ${totals.total.toFixed(2)}`,
      `Total Reimbursed,BDT ${totals.reimbursed.toFixed(2)}`,
      `Total Pending,BDT ${(totals.total - totals.reimbursed).toFixed(2)}`,
      `Reimbursement Rate,${((totals.reimbursed / totals.total) * 100).toFixed(2)}%`,
      '',
      'DETAILED EXPENSE RECORDS',
      'Date,Project Code,Project Name,Member,Category,Description,Amount (BDT),Status,Reimbursed From',
    ];
    
    expenses.forEach(e => {
      csvLines.push([
        fmtDate(e.expense_date),
        e.project_code,
        e.project_name,
        e.submitted_by_name,
        CAT_LABELS[e.category],
        `"${e.description}"`,
        Number(e.amount).toFixed(2),
        e.reimbursed ? 'Reimbursed' : 'Pending',
        e.reimbursed_from || '—',
      ].join(','));
    });
    
    csvLines.push('');
    csvLines.push(`TOTAL,,,,,,,BDT ${totals.total.toFixed(2)},`);
    csvLines.push('');
    csvLines.push('Report generated by Research Expense Tracker');
    csvLines.push('Daffodil International University - Faculty of Graduate Studies');
    
    const csv = csvLines.join('\n');
    const a = document.createElement('a');
    a.href = URL.createObjectURL(new Blob([csv], { type: 'text/csv;charset=utf-8;' }));
    a.download = `expenses-report-${dateStr}.csv`;
    a.click();
  };
